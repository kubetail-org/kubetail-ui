export const rgbToHex = (rgb) => {
  const [r, g, b] = rgb.split(' ');
  return '#' + [r, g, b].map(x => parseInt(x).toString(16).padStart(2, '0')).join('');
};

export function twValue(key) {
  const raw = getComputedStyle(document.documentElement)
    .getPropertyValue(key)
    .trim();

  // Convert "48rem" or "768px" â†’ px (number)
  if (raw.endsWith('rem')) {
    const root = parseFloat(getComputedStyle(document.documentElement).fontSize); // default 16
    return `${parseFloat(raw) * root}px`;
  }
  return `${parseFloat(raw)}px`;
}

# Tailwind Configuration

To add Kubetail-UI's custom Tailwind theme to your project with Tailwind v4, add the Tailwind plugin to your Vite config:

```typescript
import tailwindcss from '@tailwindcss/vite';
import { defineConfig } from 'vite';

export default defineConfig({
  plugins: [
    tailwindcss(),
    // other plugins...
  ],
});
```

Then import Tailwind CSS and the kubetail-ui styles in your main CSS file:

```css
@import "tailwindcss";
@import "@kubetail/ui";
```
