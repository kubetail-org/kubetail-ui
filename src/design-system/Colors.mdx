## Text Colors

export const TextColorTable = () => {
  const tuples = [
    ['background', 'foreground'],              // bg-background text-foreground
    ['card', 'card-foreground'],               // bg-card text-foreground
    ['popover', 'popover-foreground'],          // bg-popover text-popover-foreground
    ['primary', 'primary-foreground'],         // bg-primary text-primary-foreground
    ['secondary', 'secondary-foreground'],     // bg-secondary text-secondary-foreground
    ['muted', 'muted-foreground'],             // bg-muted text-muted-foreground
    ['accent', 'accent-foreground'],           // bg-accent text-accent-foreground
    ['destructive', 'destructive-foreground'], // bg-destructive text-destructive-foreground
  ];

  return (
    <Unstyled>
      <table>
        <thead>
          <tr>
            <th className="text-left">Name</th>
            <th>Light</th>
            <th>Dark</th>
          </tr>
        </thead>
        <tbody>
          {tuples.map((tuple) => (
            <tr key={tuple[0]}>
              <td className="text-sm" width={colWidth}>{tuple[1]}</td>
              <td className="light bg-background py-[20px]" width={colWidth} align="center">
                <span className={cn('p-[10px] border border-dotted border-gray-300', `bg-${tuple[0]}`, `text-${tuple[1]}`)}>
                  Example text
                </span>
              </td>
              <td className="dark dark:bg-background py-[20px]" width={colWidth} align="center">
                <span className={cn('dark p-[10px] border border-dotted border-gray-700 inline-block', `bg-${tuple[0]}`, `text-${tuple[1]}`)}>
                  Example text
                </span>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </Unstyled>
  );
};

<TextColorTable />
